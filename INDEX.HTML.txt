/* =========================================================

========================================================= */
function normKey(s){
  return String(s||"").trim().toLowerCase()
    .normalize("NFD").replace(/[\u0300-\u036f]/g,"")
    .replace(/[^\w\s]/g," ").replace(/\s+/g," ").trim();
}
function idxOf(headers, name){
  const h = headers.map(normKey);
  return h.indexOf(normKey(name));
}
function parsePtNumber(v){
  if(v==null) return null;
  let s = String(v).trim();
  if(!s) return null;
  s = s.replace(/\./g,"").replace(",",".");
  const n = Number(s);
  return Number.isFinite(n) ? n : null;
}
function parsePercent(v){
  if(v==null) return null;
  let s = String(v).trim();
  if(!s) return null;
  s = s.replace("%","").trim();
  return parsePtNumber(s); // 0..100
}
function parseHMS(v){
  if(v==null) return null;
  const s = String(v).trim();
  const m = s.match(/^(\d{1,3}):(\d{2}):(\d{2})$/);
  if(!m) return null;
  return (+m[1])*3600 + (+m[2])*60 + (+m[3]);
}
function labelFromDateBR(d){
  const s = String(d||"").trim();
  const m = s.match(/^(\d{2})\/(\d{2})\/(\d{4})$/);
  return m ? `${m[1]}/${m[2]}` : s;
}

/* =========================================================
   Fetch
========================================================= */
async function fetchJson(url){
  const controller = new AbortController();
  const to = setTimeout(() => controller.abort(), CONFIG.fetchTimeoutMs);

  try{
    const u = new URL(url);
    if(CONFIG.token) u.searchParams.set("token", CONFIG.token);
    u.searchParams.set("_", String(Date.now())); // cache bust

    const resp = await fetch(u.toString(), {
      cache: "no-store",
      signal: controller.signal,
      headers: { "Accept": "application/json,text/plain,*/*" }
    });
    if(!resp.ok) throw new Error(`HTTP ${resp.status}`);
    const json = await resp.json();

    if(json && json.ok === false) throw new Error(json.error || "API error");
    if(!json || !Array.isArray(json.headers) || !Array.isArray(json.rows)) {
      throw new Error("Formato inválido: precisa headers[] e rows[][]");
    }
    return json;
  } finally {
    clearTimeout(to);
  }
}

/* =========================================================
   Transform: payload -> séries numéricas p/ gráficos
========================================================= */
function transform(payload){
  const headers = payload.headers;
  const rows = payload.rows;

  const cDate = idxOf(headers, "data");
  const cUtil = idxOf(headers, "utilização de Maquina");
  const cEf5  = idxOf(headers, "Eficiencia -5 dias média");
  const cPcs  = idxOf(headers, "Peças Fabric.");
  const cTc   = idxOf(headers, "TC MEDIO");
  const cFunc = idxOf(headers, "Funcionando");
  const cPar  = idxOf(headers, "Parado");

  if(cDate < 0) throw new Error("Coluna 'data' não encontrada");

  const out = [];
  for(const r of rows){
    const dateBr = r[cDate];
    out.push({
      label: labelFromDateBR(dateBr),
      util:  cUtil >= 0 ? parsePercent(r[cUtil]) : null,
      ef5:   cEf5  >= 0 ? parsePercent(r[cEf5])  : null,
      pecas: cPcs  >= 0 ? parsePtNumber(r[cPcs]) : null,
      tc_s:  cTc   >= 0 ? parseHMS(r[cTc])       : null,
      func_h: cFunc>= 0 ? (parseHMS(r[cFunc]) ?? null) : null,
      par_h:  cPar >= 0 ? (parseHMS(r[cPar])  ?? null) : null
    });
  }

  // a API costuma vir mais recente -> mais antigo; para o eixo ficar natural:
  const sliced = out.slice(0, CONFIG.lastDays).reverse();

  for(const x of sliced){
    if(x.func_h!=null) x.func_h = Math.round((x.func_h/3600)*100)/100;
    else x.func_h = null;

    if(x.par_h!=null) x.par_h = Math.round((x.par_h/3600)*100)/100;
    else x.par_h = null;

    if(x.util!=null) x.util = Math.round(x.util*100)/100;
    if(x.ef5!=null)  x.ef5  = Math.round(x.ef5*100)/100;
  }
  return sliced;
}

let SCREENS = [];

let slideTimer = null;

function ensureChart(){
  if(DOM.chart) return;
  DOM.chart = echarts.init(DOM.chartEl, null, { renderer:"canvas" });
  window.addEventListener("resize", () => DOM.chart && DOM.chart.resize());

function updateHeader(){
  DOM.title.textContent = SCREENS[index]?.title || "Sem dados";
  DOM.badge.textContent = `${SCREENS.length ? (index+1) : 0}/${SCREENS.length}`;
}
function render(){
  if(!SCREENS.length) return;
  hideOverlay();
  updateHeader();
  DOM.chart.setOption(SCREENS[index].option(), true);
  slideStart = Date.now();
}
function next(){
  if(!SCREENS.length) return;
  index = (index + 1) % SCREENS.length;
  render();
}
function prev(){
  if(!SCREENS.length) return;
  index = (index - 1 + SCREENS.length) % SCREENS.length;
  render();
}
function startLoops(){
  if(progressTimer) clearInterval(progressTimer);
  progressTimer = setInterval(() => {
    if(!SCREENS.length) return;
    const p = Math.max(0, Math.min(1, (Date.now()-slideStart)/CONFIG.slideIntervalMs));
    DOM.progress.style.transform = `scaleX(${p})`;
  }, CONFIG.progressTickMs);

  if(slideTimer) clearInterval(slideTimer);
  slideTimer = setInterval(() => {
    if(!SCREENS.length) return;
    if(Date.now()-slideStart >= CONFIG.slideIntervalMs) next();
  }, 250);
}

/* =========================================================
   Load all sources (com cache localStorage)
========================================================= */
function cacheKey(i){ return `mota_tv_cache_payload_${i}`; }

async function loadAll(){
  showOverlay("Carregando dashboards…", "Buscando dados nas APIs", "");
  setStatus("partial", "Buscando dados…");

  const tasks = CONFIG.sources.map((s,i) =>
    fetchJson(s.apiUrl).then(
      payload => ({ ok:true, payload, i }),
      err => ({ ok:false, err, i })
    )
  );

  const res = await Promise.all(tasks);

  let okCount = 0;
  const seriesBySource = [];

  for(const r of res){
    if(r.ok){
      okCount++;
      try{ localStorage.setItem(cacheKey(r.i), JSON.stringify(r.payload)); }catch(_){}
      seriesBySource[r.i] = transform(r.payload);
    }else{
      // fallback cache
      let cached = null;
      try{
        const raw = localStorage.getItem(cacheKey(r.i));
        if(raw) cached = JSON.parse(raw);
      }catch(_){}
      if(cached && Array.isArray(cached.headers) && Array.isArray(cached.rows)){
        try{ seriesBySource[r.i] = transform(cached); }
        catch(_){ seriesBySource[r.i] = []; }
      }else{
        seriesBySource[r.i] = [];
      }
    }
  }

  const when = new Date().toLocaleString("pt-BR");
  if(okCount === CONFIG.sources.length) setStatus("good", `Ao vivo • ${when}`);
  else if(okCount > 0) setStatus("partial", `Parcial (${okCount}/${CONFIG.sources.length}) • ${when}`);
  else setStatus("offline", `Offline • ${when}`);

  return seriesBySource;
}

async function buildAndStart(isRefresh=false){
  ensureChart();

  const seriesBySource = await loadAll();
  const screens = [];

  for(let i=0;i<CONFIG.sources.length;i++){
    const s = CONFIG.sources[i];
    const series = seriesBySource[i] || [];
    if(series.length) screens.push(...screensForSource(s.name, series));
  }

  if(!screens.length){
    showOverlay("Sem dados para exibir", "Nenhuma fonte retornou dados válidos", "Verifique token/links e se a API retorna headers[] e rows[][].");
    DOM.title.textContent = "Sem dados";
    DOM.badge.textContent = "0/0";
    return;
  }

  SCREENS = screens;
  if(index >= SCREENS.length) index = 0;

  render();
  startLoops();

  if(refreshTimer) clearInterval(refreshTimer);
  refreshTimer = setInterval(() => {
    buildAndStart(true).catch(err => {
      showOverlay("Erro ao atualizar", String(err?.message || err), "");
      setStatus("offline", "Erro ao atualizar");
    });
  }, CONFIG.refreshMs);
}
